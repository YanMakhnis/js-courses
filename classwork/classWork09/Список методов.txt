jQuery(селектор) - поиск элементов по селектору
.children(селектор) - поиск среди дочерних элементов, совпавших по селектору
.find(селектор) - поиск среди потомков элементов, совпавших по селектору
.parent() - получить родительский элемент
.closest(селектор) - поиск среди предков, по селектору, первого совпавшего элемента при всплытии
.filter(селектор) - отфильтрует из найденных элементов элемент с нужным селектором.

.show() - Отображает каждый из совпавших элементов набора
.hide() - Скрывает каждый из совпавших элементов набора
.toggle() - Переключает режим отображения каждого из элементов набора
.fadeIn() - Делает видимыми все элементы набора, используя анимацию изменения прозрачности элементов с 0 до 1
.fadeOut() - Делает невидимыми все элементы набора, используя анимацию изменения прозрачности элементов с 1 до 0, затем устанавливает CSS свойство display в «none»

.css(имя) - Возвращает значение указанного свойства стиля для первого элемента из набора
.css(имя, значение) - Устанавливает значение указанного свойства стиля для всех элементов набора

.addClass(имя) - Добавляет указанный(е) класс(ы) к каждому элементу набора
.removeClass(имя) - Удаляет все или указанный(е) класс(ы) из набора элементов
.toggleClass(имя) - Добавляет указанный класс к элементу, если его нет, или удаляет, если элемент уже обладает таковым
.hasClass(имя) - Возвращает true, если хотя бы один из набора элементов обладает указанным классом

.attr(имя) - Получение значения свойства первого элемента из набора
.attr(имя, значение) - Изменяет значение свойства для каждого элемента из набора
.removeAttr(имя) - Удаляет указанный атрибут у каждого элемента набора

.text() - Получает текстовое содержимое всех элементов набора, результатом является строка, содержащая объединенный текст всех совпавших элементов
.text(значение) - Вставка текста во все элементы набора

.html() - Получает содержимое HTML (innerHTML) каждого элемента набора
.html(значение) - Меняет код HTML (innerHTML) для каждого элемента набора

.val() - Получает содержимое атрибута value для первого элемента в наборе
.val(значение) - Устанавливает значение атрибута value для каждого элемента в наборе

.width() - Получает текущее значение ширины в пикселях для первого элемента в наборе
.height() - Получает текущее значение высоты в пикселях для первого элемента в наборе
.offset() - Получает текущие значения отступов относительно документа для первого элемента в наборе
.position() - Получает значение позиции элемента сверху и слева относительно его «родителя»
.scrollTop() - Получает значение отступа прокрутки от верхнего края экрана для первого элемента в наборе

.after(HTML код) - Вставляет контент после каждого элемента набора
.before(HTML код) - Вставляет контент перед каждым элементом набора
.append(HTML код) - Добавляет контент внутрь каждого элемента набора (после существующего контента, как последний дочерний элемент)
.prepend(HTML код) - Добавляет контент внутрь каждого элемента набора (перед существующим контентом, как первый дочерний элемент)
.wrap(HTML код) - Вставляет каждый совпавший элемент набора в указанную конструкцию HTML кода (добавляет внешнюю HTML обёртку)
.wrapInner(HTML код) - оборачивает контент совпавшего элемента в указанную конструкцию HTML кода (добавляет внутреннюю HTML обёртку)
.remove() - Удаляет все выбранные элементы из DOM
.detach() - Удаляет все выбранные элементы из DOM, не удаляя привязанные к ним события и информацию о элементе, и поэтому он, в целом, может быть восстановлен

.each(function) - Перебирает по очереди элементы в наборе jQuery,
запуская функцию function для каждого из них. Внутри функции текущий
элемент доступен через this, поэтому можете использовать $(this)

.eq(index) - Возвращает элемент с номером index из набора jQuery (счёт начинается с 0)
.index() - Возвращает индекс заданного элемента относительно соседних элементов (порядковый номер ребёнка у его родителя)

.on(eventName, [selector], function) - добавляет обработчик события function, для события eventName. Опционально можно добавить параметр selector, для делегирования события
.off(eventName, [selector], function) - удаляет обработчик события
.trigger(eventName) - даёт браузеру что на текущем элементе произошло событие eventName
